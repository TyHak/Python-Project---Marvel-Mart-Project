dicRegions = {}
for index, row in CleanSP.iterrows():
    if row.loc['Region'] in dicRegions:
        dicRegions[row.loc['Region']].add(row.loc['Country'])
    else:
        dicRegions[row.loc['Region']] = set()
        dicRegions[row.loc['Region']].add(row.loc['Country'])
